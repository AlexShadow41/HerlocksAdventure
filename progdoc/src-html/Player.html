001import java.util.Stack;
002import java.util.HashMap;
003
004/**
005 * Gère les actions et les données relative à un joueur
006 *
007 * @author Alexian Plancke
008 * @version Intermédiaire
009 */
010public class Player
011{
012    private String aCurrentRoom;
013    private Stack <String> aPreviousRooms = new Stack<String>();
014    private String aPlayerName;
015    private ItemList aInventory = new ItemList();
016    private int aStrength;
017    private int aAgility;
018    private int aWisdom;
019    private int aIntelligence;
020    private float aCarriedWeight;
021    private float aExtraWeigth;
022    
023
024    /**
025     * Constructeur d'objets de classe Player
026     */
027    public Player()
028    {
029        this.aPlayerName = "Player 1";
030        this.aCurrentRoom = "Road1";
031        this.aStrength = 10 + (int) (Math.random() * 10)  ;
032        this.aAgility = 5 + (int) (Math.random() * 15);
033        this.aWisdom = 5 + (int) (Math.random() * 15);
034        this.aIntelligence = 10 + (int) (Math.random() * 10);
035        this.aExtraWeigth = this.aCarriedWeight = 0;
036    }
037    
038    /**
039     * Enregistre le nom du joueur
040     * @param pPlayerName String Nom du joueur
041     */
042    public void setName(final String pPlayerName){
043        this.aPlayerName = pPlayerName;
044    }
045    
046    /**
047     * Permet d'accéder à la pièce où se trouve le joueur
048     * @param pRooms HashMap Dictionnaire des salles du jeu
049     * @return       Renvoie l'objet Room de la salle dans laquelle se trouve le joueur
050     */
051    public Room getCurrentRoom(final HashMap<String, Room> pRooms){
052        return pRooms.get(this.aCurrentRoom);
053    }
054    
055    /**
056     * Initialise la position initiale du joueur
057     * @param pRoomName String Nom de la Salle
058     */
059    public void setCurrentRoom(final String pRoomName){
060        // this.aPreviousRooms.push(this.aCurrentRoom);
061        this.aCurrentRoom = pRoomName;
062    }
063    
064    /**
065     * 
066     * @param pRooms HashMap Liste des Salles
067     * @param pDirection String La direction
068     * @return Renvoie le nom de la nouvelle salle courante
069     */
070    public String go(final HashMap<String,Room> pRooms, final String pDirection)
071    {
072        if (pDirection.equals("back")) {
073            if (this.aPreviousRooms.isEmpty())
074                return null;
075            this.aCurrentRoom = this.aPreviousRooms.pop();
076        }
077        else {
078            Room vRoom = pRooms.get(this.aCurrentRoom);
079            String vNextRoom = vRoom.getExits(pDirection);
080            if (vNextRoom == null)
081                return null;
082            this.aPreviousRooms.push(this.aCurrentRoom);
083            this.aCurrentRoom = vNextRoom;
084        }
085        return(this.aCurrentRoom);
086    }
087    
088    /**
089     * Fonction permettant dau joueur de prendre un item
090     * @param pRooms HashMap Liste des Salles
091     * @param   pItemName String Nom de l'item à ajouter à l'iventaire du joueur
092     * @return  Renvoie le message à afficher pour informer le joueur du résultat de l'action
093     */
094    public String take(final HashMap<String,Room> pRooms, final String pItemName) {
095        // Can the requested item be found in the current room?
096        Room vRoom = pRooms.get(this.aCurrentRoom);
097        Item vItem = vRoom.takeItem(pItemName);
098        if (vItem == null)
099            return "There's no such item in this room";
100        
101        // Can the player carry all the weight?
102        double vCanCarry = this.aStrength * 2.5 * 1000.0 + this.aExtraWeigth;
103        double vToCarry = this.aCarriedWeight + vItem.getWeight();
104        if (vToCarry > vCanCarry) {
105            vRoom.addItem(pItemName, vItem);
106            return "You are not strong enough to carry this, maybe you should consider droping some stuff";
107        }//if
108        // Add the item to player's inventory
109        this.aInventory.addItem(pItemName, vItem);
110        this.aCarriedWeight += vItem.getWeight();
111        return "The "+pItemName+" has been added to your inventory";
112    }
113
114    /**
115     * Méthode permettant au joueur d'inspecter un objet (A améliorer plus tard avec la loupe)
116     * @param pRooms HashMap Liste des Salles
117     * @param pItemName String L'item à inspecter
118     * @return  Renvoie le message à afficher pour informer le joueur du résultat de l'action
119     */
120    public String inspect(final HashMap<String,Room> pRooms, final String pItemName) {
121        // Can the requested item be found in the current room?
122        Room vRoom = pRooms.get(this.aCurrentRoom);
123        Item vItem = vRoom.findItem(pItemName);
124        if (vItem == null)
125            return "There's no such item in this room";
126        return vItem.getDescription();
127        //return vItem.getLongDescription();
128    }
129
130    
131    /**
132     * Méthode permettant au joueur de manger
133      * @param pRooms HashMap Liste des Salles
134     * @param pItemName String L'item à manger
135     * @return  Renvoie le message à afficher pour informer le joueur du résultat de l'action
136     */
137    public String eat(final HashMap<String,Room> pRooms, final String pItemName) {
138        // Check if in your inventory
139        if (! this.aInventory.searchItem(pItemName))
140            return "You don't have any "+pItemName+" in your inventory";
141
142        // check if you can eat it
143        if (! pItemName.equals("Cookie"))
144            return "Oh no, this doesn't look tasty, try it yourself!";
145
146        // Remove it from your inventory
147        Item vItem = this.aInventory.removeItem(pItemName);
148        this.aCarriedWeight -= vItem.getWeight();
149        
150        // Apply magic effect
151        this.aExtraWeigth = 60*1000; 
152        return "Oh my god, I now feel so strong";
153    }
154
155    /**
156     * Méthode permettant au joueur de reposer un objet
157     * @param pRooms HashMap Liste des Salles
158     * @param pItemName String L'item à déposer
159     * @return  Renvoie le message à afficher pour informer le joueur du résultat de l'action
160     */
161    public String drop(final HashMap<String,Room> pRooms, final String pItemName) {
162        if (! this.aInventory.searchItem(pItemName))
163            return "You don't have any "+pItemName+" in your inventory";
164        Item vItem = this.aInventory.removeItem(pItemName);
165        this.aCarriedWeight -= vItem.getWeight();
166
167        // Put the dropped Item in the current room
168        Room vRoom = pRooms.get(this.aCurrentRoom);
169        vRoom.addItem(pItemName, vItem);
170        return "The "+pItemName+" has been dropped";
171    }
172    
173    /**
174     * Renvoie l'inventaire du joueur sous forme de String
175     * @return  La chaine contenant la description de l'inventaire
176     */
177    public String inventory() {
178        String vInventory = this.aInventory.listeItems();
179        if (vInventory.equals(""))
180            return "Empty";
181        return vInventory;
182    }

184}
