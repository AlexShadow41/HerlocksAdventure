001import java.util.HashMap;
002import java.util.Set;
003
004/**
005 * Classe qui sert représenter chacunes des pièces que le joueur peut visiter dans le jeu
006 *
007 * @author Alexian Plancke
008 * @version Intermédiaire
009 * 
010 */
011public class Room
012{
013    private String aDescription;
014    private HashMap<String, String> aExits;
015    public static final Room UNKNOWN_DIRECTION = new Room( "nowhere", "");
016    private String aImageName;
017    private ItemList aItems;
018    
019    /**
020     * Constructeur naturel
021     * @param pD String Description de la salle
022     * @param pI String Nom de l'image
023    */
024    public Room (final String pD, final String pI)
025    {
026        this.aDescription = pD;
027        this.aExits = new HashMap<String, String>();
028        this.aItems = new ItemList();
029        this.aImageName = pI;
030    } //Room
031    
032    /**
033     * Retourne la description d'une pièce
034     * @return      La description de la pièce
035    */
036    public String getDescription()
037    {
038        return this.aDescription;
039    } //getDescription
040    
041    /**
042     * Permet d'ajouter un Item (Avec description précise)
043     * @param pObjName String Nom de l'Item
044     * @param pObjD String Description de l'Item
045     * @param pObjW int Poids de l'Item
046     * @param pObjLD String Description précise de l'Item
047    */
048    public void addItem(final String pObjName, final String pObjD, final int pObjW, final String pObjLD)
049    {
050        Item It = new Item(pObjD, pObjW);
051        It.setLongDescription(pObjLD);
052        this.aItems.addItem(pObjName, It);
053    }
054    
055    /**
056     * Permet d'ajouter un Item
057     * @param pObjName String Nom de l'Item
058     * @param pObjD String Description de l'Item
059     * @param pObjW int Poids de l'Item
060    */
061    public void addItem(final String pObjName, final String pObjD, final int pObjW)
062    {
063        Item vItem = new Item(pObjD, pObjW);
064        this.aItems.addItem(pObjName, vItem);
065    }
066
067    /**
068     * Permet d'ajouter un Item
069     * @param pObjName String Nom de l'Item
070     * @param pItem Item Objet à ajouter
071    */
072    public void addItem(final String pObjName, final Item pItem)
073    {
074        this.aItems.addItem(pObjName, pItem);
075    }
076    
077    /**
078     * Retourne la liste des items d'une pièce
079     * @return      La liste des objets présent dans la pièce
080    */
081    public String getItemList()
082    {
083        String vS = this.aItems.listeItems();
084        if (vS != "")
085            return "You can see some items: " + vS + "\n";
086        return "";
087    } //getDescription
088
089    /**
090     * Supprime un objet d'une salle et le renvoie pour être ajouté à l'inventaire du joueur
091     * Retourne null si l'article est introuvable dans la pièce
092     * @param       pItemName   Nom de l'item 
093     * @return      L'objet Item pris dans la pièce
094    */
095    public Item takeItem(final String pItemName)
096    {
097        return this.aItems.removeItem(pItemName);
098    }
099
100    /**
101     * Rechercher un élément dans une pièce 
102     * Retourne null si l'article est introuvable dans la pièce
103     * @param       pItemName   Nom de l'item 
104     * @return      L'objet Item pris dans la pièce
105    */
106    public Item findItem(final String pItemName)
107    {
108        return this.aItems.findItem(pItemName);
109    }
110    
111    /**
112     * Retourne la description détaillée d'une pièce
113     * @return      La description détaillée de la pièce
114    */
115    public String getLongDescription()
116    {
117        return "You are " + this.aDescription + ".\n" + this.getLocationInfo();
118    } //getDescription
119
120    /**
121     * Retourne la chemin vers l'image d'une pièce
122     * @return      Le chemin d'accès à l'image
123    */
124    public String getImageName()
125    {
126        return this.aImageName;
127    }
128    
129    /**
130     * Définis les sorties
131     * @param pNorthExit Room Sortie Nord
132     * @param pSouthExit Room Sortie Sud
133     * @param pEastExit Room Sortie Est
134     * @param pWestExit Room Sortie Ouest
135    */
136    public void setExits(final String pNorthExit,final String pSouthExit, final String pEastExit, final String pWestExit)
137    {
138        if (pNorthExit != "")
139            this.aExits.put("north", pNorthExit);
140        if (pSouthExit != "")
141            this.aExits.put("south", pSouthExit);
142        if (pEastExit != "")
143            this.aExits.put("east", pEastExit);
144        if (pWestExit != "")
145            this.aExits.put("west", pWestExit);
146    }
147    
148    /**
149     * Ajoute une sortie
150     * @param pDirection String La direction à rajouter
151     * @param pExit Room Room La pièce qui se trouve dans cette direction  
152    */
153    public void setExit(final String pDirection, final String pExit)
154    {
155        this.aExits.put(pDirection, pExit);
156    }
157     /*public void setExit(final String pDirection, final Room pExit)
158    {
159        this.exits.put(pDirection, pExit);
160    }*/
161    
162    
163    /**
164     * Acceseur sur une sortie d'une salle en fonction de la direction fournie
165     * @param pDirection String La direction fournie
166     * @return La sortie dans cette direction
167    */
168    public String getExits(String pDirection)
169    {
170        return(this.aExits.get(pDirection));
171    }
172    
173   
174    /**
175     * Renvoie la liste des sorties d'une salle sous forme d'une chaine de caractère
176     * @return      Retourne la liste des sorties d'une pièce
177    */
178    public String getLocationInfo()
179    {
180        String ReturnString = "Exits: [";
181        Set<String> keys = this.aExits.keySet();
182        for(String exit : keys) {
183            ReturnString += " " + exit;
184        }
185        ReturnString += "]";
186        return(ReturnString);
187   
188    }
189
190}// Room
