008/**
009 * This class is part of the "World of Zuul" application. 
010 * "World of Zuul" is a very simple, text based adventure game.  
011 * 
012 * This class holds an enumeration table of all command words known to the game.
013 * It is used to recognise commands as they are typed in.
014 *
015 * @author  Michael Kolling and David J. Barnes + D.Bureau
016 * @version 2008.03.30 + 2019.09.25
017 */
018public class CommandWords
019{
020    // a constant array that will hold all valid command words
021    private static String[] aValidCommands = { "go", "help", "look", "take", "drop", "smoke", "inventory", "quit"};
022
023    /**
024     * Constructor - initialise the command words.
025     */
026    public CommandWords()
027    {
028    } // CommandWords()
029
030    /**
031     * Check whether a given String is a valid command word. 
032     * @return true if a given string is a valid command,
033     * false if it isn't.
034     * @param pString String La commande à analyser
035     */
036    public boolean isCommand( final String pString )
037    {
038        for ( int vI=0; vI<CommandWords.aValidCommands.length; vI++ ) {
039            if ( CommandWords.aValidCommands[vI].equals( pString ) )
040                return true;
041        } // for
042        // if we get here, the string was not found in the commands :
043        return false;
044    } // isCommand()
045    
046    /**
047     * Affiche l'ensemble des commandes valides du jeu
048     * @return La chaine qui conrient les commandes du jeu
049    */
050    public static String getCommandList(){
051        String vS = "";
052        for(String command: CommandWords.aValidCommands){
053            vS += command + " ";
054        }
055        return(vS);
056    }
057} // CommandWords
